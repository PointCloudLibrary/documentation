146  * \brief macro for compatibility across compilers and help remove old deprecated
147  * items for the Major.Minor release

149  * \details compiler errors of `unneeded_deprecation` and `major_version_mismatch`
150  * are hints to the developer that those items can be safely removed.

385 inline void*
386 aligned_malloc(std::size_t size)
387 {
388  void* ptr;
389 #if defined(MALLOC_ALIGNED)
390  ptr = std::malloc(size);
391 #elif defined(HAVE_POSIX_MEMALIGN)
392  if (posix_memalign(&ptr, 16, size))
393  ptr = 0;
394 #elif defined(HAVE_MM_MALLOC)
395  ptr = _mm_malloc(size, 16);
396 #elif defined(_MSC_VER)
397  ptr = _aligned_malloc(size, 16);
398 #elif defined(ANDROID)
399  ptr = memalign(16, size);
400 #else
401 #error aligned_malloc not supported on your platform
402  ptr = 0;
403 #endif
404  return (ptr);
405 }

456 // [[unlikely]] can be used on any conditional branch, but __builtin_expect is restricted to the evaluation point
457 // This makes it quite difficult to create a single macro for switch and while/if
458 /**
459  * @def PCL_CONDITION_UNLIKELY
460  * @brief Tries to inform the compiler to optimize codegen assuming that the condition will probably evaluate to false
461  * @note Prefer using `PCL_{IF,WHILE}_UNLIKELY`
462  * @warning This can't be used with switch statements
463  * @details This tries to help the compiler optimize for the unlikely case.
464  * Most compilers assume that the condition would evaluate to true in if and while loops (reference needed)
465  * As such the opposite of this macro (PCL_CONDITION_LIKELY) will not result in significant performance improvement
466  *
467  * Some sample usage:
468  * @code{.cpp}
469  * if PCL_CONDITION_UNLIKELY(x == 0) { return; } else { throw std::runtime_error("some error"); }
470  * //
471  * while PCL_CONDITION_UNLIKELY(wait_for_result) { sleep(1); } // busy wait, with minimal chances of waiting
472  * @endcode
473  */
474 #if __has_cpp_attribute(unlikely)
475  #define PCL_CONDITION_UNLIKELY(x) (static_cast<bool>(x)) [[unlikely]]
476 #elif defined(__GNUC__)
477  #define PCL_CONDITION_UNLIKELY(x) (__builtin_expect(static_cast<bool>(x), 0))
478 #elif defined(__clang__) && (PCL_LINEAR_VERSION(__clang_major__, __clang_minor__, 0) >= PCL_LINEAR_VERSION(3, 9, 0))
479  #define PCL_CONDITION_UNLIKELY(x) (__builtin_expect(static_cast<bool>(x), 0))
480 #else // MSVC has no such alternative
481  #define PCL_CONDITION_UNLIKELY(x) (x)
482 #endif
483 
484 #define PCL_IF_UNLIKELY(x) if PCL_CONDITION_UNLIKELY(x)
485 #define PCL_WHILE_UNLIKELY(x) while PCL_CONDITION_UNLIKELY(x)
