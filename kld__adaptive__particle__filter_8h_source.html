150  /** \brief return upper quantile of standard normal distribution.
151  * \param[in] u ratio of quantile.
152  */
153  double
154  normalQuantile(double u)
155  {
156  const double a[9] = {1.24818987e-4,
157  -1.075204047e-3,
158  5.198775019e-3,
159  -0.019198292004,
160  0.059054035642,
161  -0.151968751364,
162  0.319152932694,
163  -0.5319230073,
164  0.797884560593};
165  const double b[15] = {-4.5255659e-5,
166  1.5252929e-4,
167  -1.9538132e-5,
168  -6.76904986e-4,
169  1.390604284e-3,
170  -7.9462082e-4,
171  -2.034254874e-3,
172  6.549791214e-3,
173  -0.010557625006,
174  0.011630447319,
175  -9.279453341e-3,
176  5.353579108e-3,
177  -2.141268741e-3,
178  5.35310549e-4,
179  0.999936657524};
180  double w, y, z;
181 
182  if (u == 0.)
183  return (0.5);
184  y = u / 2.0;
185  if (y < -3.)
186  return (0.0);
187  if (y > 3.)
188  return (1.0);
189  if (y < 0.0)
190  y = -y;
191  if (y < 1.0) {
192  w = y * y;
193  z = a[0];
194  for (int i = 1; i < 9; i++)
195  z = z * w + a[i];
196  z *= (y * 2.0);
197  }
198  else {
199  y -= 2.0;
200  z = b[0];
201  for (int i = 1; i < 15; i++)
202  z = z * y + b[i];
203  }
204 
205  if (u < 0.0)
206  return ((1. - z) / 2.0);
207  return ((1. + z) / 2.0);
208  }

210  /** \brief calculate K-L boundary. K-L boundary follows 1/2e*chi(k-1, 1-d)^2.
211  * \param[in] k the number of bins and the first parameter of chi distribution.
212  */
213  virtual double
214  calcKLBound(int k)
215  {
216  double z = normalQuantile(delta_);
217  double chi = 1.0 - 2.0 / (9.0 * (k - 1)) + sqrt(2.0 / (9.0 * (k - 1))) * z;
218  return ((k - 1.0) / (2.0 * epsilon_) * chi * chi * chi);
219  }
