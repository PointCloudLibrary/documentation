52  /** \brief @b GraphHandler class is a wrapper for a general SLAM graph
53  * The actual graph class must fulfill the following boost::graph concepts:
54  * - BidirectionalGraph
55  * - AdjacencyGraph
56  * - VertexAndEdgeListGraph
57  * - MutableGraph
58  *
59  * Other valid expressions:
60  * - add_edge (m,g) add a new edge according to the given measurement. Return type: std::pair<edge_descriptor, bool>
61  * - add_vertex (e,g) add a new vertex according to the given estimate. Return type: vertex_descriptor
62  * - get_pose (v,g) retrieve the pose estimate for v, if any. Return type: Eigen::Matrix4f
63  * - get_cloud (v,g) retrieve the cloud pointer associated to v, if any. Return type: pcl::PointCloud<PointT>::ConstPtr
64  * - set_estimate (v,e,g) set the estimate for an existing vertex. Return type: void.
65  * - set_measurement (d,m,g) set the measurement for an existing edge. Return type: void.
66  * Notation:
67  * - m an edge measurement
68  * - e a vertex estimate
69  * - v a vertex
70  * - d an edge
71  * - g a graph
72  * A valid graph implementation should accept at least the PoseEstimate estimate and the PoseMeasurement measurement
73  *
74  * If a specific graph implementation needs initialization and/or finalization,
75  * specialize the protected methods init() and deinit() for your graph type
76  *
77  * \author Nicola Fioraio
78  * \ingroup registration
79  */
80  template <typename GraphT>
81  class GraphHandler
82  {
83  public:
84  using Ptr = shared_ptr<GraphHandler<GraphT> >;
85  using ConstPtr = shared_ptr<const GraphHandler<GraphT> >;
86  using GraphPtr = shared_ptr<GraphT>;
87  using GraphConstPtr = shared_ptr<const GraphT>;
88 
89  using Vertex = typename boost::graph_traits<GraphT>::vertex_descriptor;
90  using Edge = typename boost::graph_traits<GraphT>::edge_descriptor;
91 
92  /** \brief Empty constructor. */
93  GraphHandler () : graph_impl_ (new GraphT ())
94  {
95  if (!init ())
96  throw InitFailedException ("Graph Initialization Failed", __FILE__, "pcl::registration::GraphHandler::GraphHandler ()", __LINE__);
97  }
98 
99  /** \brief Destructor. */
100  ~GraphHandler ()
101  {
102  deinit ();
103  }
104 
105  /** \brief Clear the graph */
106  inline void
107  clear ()
108  {
109  deinit ();
110  graph_impl_.reset (new GraphT ());
111  if (!init ())
112  throw InitFailedException ("Graph Initialization Failed", __FILE__, "pcl::registration::GraphHandler::clear ()", __LINE__);
113  }

129  /** \brief Add a new point cloud to the graph and return the new vertex
130  * \param cloud the new point cloud
131  * \param pose the pose estimate
132  * \return a reference to the new vertex
133  */
134  template <class PointT> inline Vertex
135  addPointCloud (const typename pcl::PointCloud<PointT>::ConstPtr& cloud, const Eigen::Matrix4f& pose)
136  {
137  return add_vertex (PoseEstimate<PointT> (cloud, pose), *graph_impl_);
138  }

140  /** \brief Add a new generic vertex created according to the given estimate
141  * \param estimate the parameters' estimate
142  * \return a reference to the new vertex
143  */
144  template <class EstimateT> inline Vertex
145  addGenericVertex (const EstimateT& estimate)
146  {
147  return add_vertex (estimate, *graph_impl_);
148  }

150  /** \brief Add a new constraint between two poses
151  * \param v_start the first pose
152  * \param v_end the second pose
153  * \param relative_transformation the transformation from v_start to v_end
154  * \param information_matrix the uncertainty
155  * \return a reference to the new edge
156  */
157  template <class InformationT> inline Edge
158  addPoseConstraint ( const Vertex& v_start, const Vertex& v_end,
159  const Eigen::Matrix4f& relative_transformation,
160  const InformationT& information_matrix)
161  {
162  return add_edge ( PoseMeasurement<Vertex, InformationT> (v_start, v_end, relative_transformation, information_matrix),
163  *graph_impl_);
164  }
165 
166  /** \brief Add a generic constraint created according to the given measurement
167  * \param measurement the measurement
168  * \return a reference to the new edge
169  */
170  template <class MeasurementT> inline Edge
171  addGenericConstraint (const MeasurementT& measurement)
172  {
173  return add_edge (measurement, *graph_impl_);
174  }
175 
176  /** \brief Remove a vertex from the graph
177  * \param v the vertex
178  */
179  inline void
180  removeVertex (const Vertex& v)
181  {
182  remove_vertex (v.v_, *graph_impl_);
183  }
184 
185  /** \brief Remove a constraint from the graph
186  * \param e the edge
187  */
188  inline void
189  removeConstraint (const Edge& e)
190  {
191  remove_edge(e.e_, *graph_impl_);
192  }
193 
194  protected:
195  /** \brief This method is called right after the creation of graph_impl_ */
196  inline bool
197  init ()
198  {
199  return true;
200  }
201 
202  /** \brief This method is called when graph_impl_ is going to be destroyed */
203  inline bool
204  deinit ()
205  {
206  return true;
207  }
208 
209  private:

211  };
