65  /** \brief Get the square distance from a point to a line (represented by a point and a direction)
66  * \param pt a point
67  * \param line_pt a point on the line (make sure that line_pt[3] = 0 as there are no internal checks!)
68  * \param line_dir the line direction
69  * \ingroup common
70  */
71  double inline
72  sqrPointToLineDistance (const Eigen::Vector4f &pt, const Eigen::Vector4f &line_pt, const Eigen::Vector4f &line_dir)
73  {
74  // Calculate the distance from the point to the line
75  // D = ||(P2-P1) x (P1-P0)|| / ||P2-P1|| = norm (cross (p2-p1, p1-p0)) / norm(p2-p1)
76  return (line_dir.cross3 (line_pt - pt)).squaredNorm () / line_dir.squaredNorm ();
77  }

79  /** \brief Get the square distance from a point to a line (represented by a point and a direction)
80  * \note This one is useful if one has to compute many distances to a fixed line, so the vector length can be pre-computed
81  * \param pt a point
82  * \param line_pt a point on the line (make sure that line_pt[3] = 0 as there are no internal checks!)
83  * \param line_dir the line direction
84  * \param sqr_length the squared norm of the line direction
85  * \ingroup common
86  */
87  double inline
88  sqrPointToLineDistance (const Eigen::Vector4f &pt, const Eigen::Vector4f &line_pt, const Eigen::Vector4f &line_dir, const double sqr_length)
89  {
90  // Calculate the distance from the point to the line
91  // D = ||(P2-P1) x (P1-P0)|| / ||P2-P1|| = norm (cross (p2-p1, p1-p0)) / norm(p2-p1)
92  return (line_dir.cross3 (line_pt - pt)).squaredNorm () / sqr_length;

95  /** \brief Obtain the maximum segment in a given set of points, and return the minimum and maximum points.
96  * \param[in] cloud the point cloud dataset
97  * \param[out] pmin the coordinates of the "minimum" point in \a cloud (one end of the segment)
98  * \param[out] pmax the coordinates of the "maximum" point in \a cloud (the other end of the segment)
99  * \return the length of segment length
100  * \ingroup common
101  */
102  template <typename PointT> double inline
103  getMaxSegment (const pcl::PointCloud<PointT> &cloud,
104  PointT &pmin, PointT &pmax)
105  {
106  double max_dist = std::numeric_limits<double>::min ();
107  const auto token = std::numeric_limits<std::size_t>::max();
108  std::size_t i_min = token, i_max = token;
109 
110  for (std::size_t i = 0; i < cloud.points.size (); ++i)
111  {
112  for (std::size_t j = i; j < cloud.points.size (); ++j)
113  {
114  // Compute the distance
115  double dist = (cloud.points[i].getVector4fMap () -
116  cloud.points[j].getVector4fMap ()).squaredNorm ();
117  if (dist <= max_dist)
118  continue;
119 
120  max_dist = dist;
121  i_min = i;
122  i_max = j;
123  }
124  }
125 
126  if (i_min == token || i_max == token)
127  return (max_dist = std::numeric_limits<double>::min ());
128 
129  pmin = cloud.points[i_min];
130  pmax = cloud.points[i_max];
131  return (std::sqrt (max_dist));
132  }

174  /** \brief Calculate the squared euclidean distance between the two given points.
175  * \param[in] p1 the first point
176  * \param[in] p2 the second point
177  */
178  template<typename PointType1, typename PointType2> inline float
179  squaredEuclideanDistance (const PointType1& p1, const PointType2& p2)
180  {
181  float diff_x = p2.x - p1.x, diff_y = p2.y - p1.y, diff_z = p2.z - p1.z;
182  return (diff_x*diff_x + diff_y*diff_y + diff_z*diff_z);
183  }
184 
185  /** \brief Calculate the squared euclidean distance between the two given points.
186  * \param[in] p1 the first point
187  * \param[in] p2 the second point
188  */
189  template<> inline float
190  squaredEuclideanDistance (const PointXY& p1, const PointXY& p2)
191  {
192  float diff_x = p2.x - p1.x, diff_y = p2.y - p1.y;
193  return (diff_x*diff_x + diff_y*diff_y);
194  }

196  /** \brief Calculate the euclidean distance between the two given points.
197  * \param[in] p1 the first point
198  * \param[in] p2 the second point
199  */
200  template<typename PointType1, typename PointType2> inline float
201  euclideanDistance (const PointType1& p1, const PointType2& p2)
202  {
