46 template <typename Point1T, typename Point2T> void
47 pcl::getApproximateIndices (
48  const typename pcl::PointCloud<Point1T>::ConstPtr &cloud_in,
49  const typename pcl::PointCloud<Point2T>::ConstPtr &cloud_ref,
50  std::vector<int> &indices)
51 {
52  pcl::KdTreeFLANN<Point2T> tree;
53  tree.setInputCloud (cloud_ref);
54 
55  std::vector<int> nn_idx (1);
56  std::vector<float> nn_dists (1);
57  indices.resize (cloud_in->points.size ());
58  for (std::size_t i = 0; i < cloud_in->points.size (); ++i)
59  {
60  tree.nearestKSearchT ((*cloud_in)[i], 1, nn_idx, nn_dists);
61  indices[i] = nn_idx[0];
62  }
63 }
