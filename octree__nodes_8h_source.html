190 
191  /** \brief Empty constructor. */
192  OctreeBranchNode(const OctreeBranchNode& source) : OctreeNode()

194  memset(child_node_array_, 0, sizeof(child_node_array_));

196  for (unsigned char i = 0; i < 8; ++i)
