108  /** \brief A helper functor that can copy a specific value if the given field exists.
109  *
110  * \note In order to actually copy the value an instance of this functor should be passed
111  * to a pcl::for_each_type loop. See the example below.
112  *
113  * \code
114  * PointInT p;
115  * bool exists;
116  * float value;
117  * using FieldList = typename pcl::traits::fieldList<PointInT>::type;
118  * pcl::for_each_type<FieldList> (pcl::CopyIfFieldExists<PointT, float> (p, "intensity", exists, value));
119  * \endcode
120  */
121  template <typename PointInT, typename OutT>
122  struct CopyIfFieldExists
123  {
124  using Pod = typename traits::POD<PointInT>::type;
125 
126  /** \brief Constructor.
127  * \param[in] pt the input point
128  * \param[in] field the name of the field
129  * \param[out] exists set to true if the field exists, false otherwise
130  * \param[out] value the copied field value
131  */
132  CopyIfFieldExists (const PointInT &pt,
133  const std::string &field,
134  bool &exists,
135  OutT &value)
136  : pt_ (reinterpret_cast<const Pod&>(pt)), name_ (field), exists_ (exists), value_ (value)
137  {
138  exists_ = false;
139  }
140 
141  /** \brief Constructor.
142  * \param[in] pt the input point
143  * \param[in] field the name of the field
144  * \param[out] value the copied field value
145  */
146  CopyIfFieldExists (const PointInT &pt,
147  const std::string &field,
148  OutT &value)
149  : pt_ (reinterpret_cast<const Pod&>(pt)), name_ (field), exists_ (exists_tmp_), value_ (value)
150  {
151  }
152 
153  /** \brief Operator. Data copy happens here. */
154  template <typename Key> inline void
155  operator() ()
156  {
157  if (name_ == pcl::traits::name<PointInT, Key>::value)
158  {
159  exists_ = true;
160  using T = typename pcl::traits::datatype<PointInT, Key>::type;
161  const std::uint8_t* data_ptr = reinterpret_cast<const std::uint8_t*>(&pt_) + pcl::traits::offset<PointInT, Key>::value;
162  value_ = static_cast<OutT> (*reinterpret_cast<const T*>(data_ptr));
163  }
164  }
165 
166  private:
167  const Pod &pt_;
168  const std::string &name_;
169  bool &exists_;
170  // Bogus entry
171  bool exists_tmp_;
172  OutT &value_;
173  };

175  /** \brief A helper functor that can set a specific value in a field if the field exists.
176  *
177  * \note In order to actually set the value an instance of this functor should be passed
178  * to a pcl::for_each_type loop. See the example below.
179  *
180  * \code
181  * PointT p;
182  * using FieldList = typename pcl::traits::fieldList<PointT>::type;
183  * pcl::for_each_type<FieldList> (pcl::SetIfFieldExists<PointT, float> (p, "intensity", 42.0f));
184  * \endcode
185  */
186  template <typename PointOutT, typename InT>
187  struct SetIfFieldExists
188  {
189  using Pod = typename traits::POD<PointOutT>::type;
190 
191  /** \brief Constructor.
192  * \param[in] pt the input point
193  * \param[in] field the name of the field
194  * \param[out] value the value to set
195  */
196  SetIfFieldExists (PointOutT &pt,
197  const std::string &field,
198  const InT &value)
199  : pt_ (reinterpret_cast<Pod&>(pt)), name_ (field), value_ (value)
200  {
201  }
202 
203  /** \brief Operator. Data copy happens here. */
204  template <typename Key> inline void
205  operator() ()
206  {
207  if (name_ == pcl::traits::name<PointOutT, Key>::value)
208  {
209  using T = typename pcl::traits::datatype<PointOutT, Key>::type;
210  std::uint8_t* data_ptr = reinterpret_cast<std::uint8_t*>(&pt_) + pcl::traits::offset<PointOutT, Key>::value;
211  *reinterpret_cast<T*>(data_ptr) = static_cast<T> (value_);
212  }
213  }
214 
215  private:
216  Pod &pt_;
217  const std::string &name_;
218  const InT &value_;
219  };
