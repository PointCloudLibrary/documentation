49  enum ColorLUTName
50  {
51  /** Color lookup table consisting of 256 colors structured in a maximally
52  * discontinuous manner. Generated using the method of Glasbey et al.
53  * (see https://github.com/taketwo/glasbey) */
54  LUT_GLASBEY,
55  /** A perceptually uniform colormap created by Stéfan van der Walt and
56  * Nathaniel Smith for the Python matplotlib library.
57  * (see https://youtu.be/xAoljeRJ3lU for background and overview) */
58  LUT_VIRIDIS,
59  };

62  class ColorLUT
63  {
64 
65  public:
66 
67  /** Get a color from the lookup table with a given id.
68  *
69  * The id should be less than the size of the LUT (see size()). */
70  static RGB at (std::size_t color_id);
71 
72  /** Get the number of colors in the lookup table.
73  *
74  * Note: the number of colors is different from the number of elements
75  * in the lookup table (each color is defined by three bytes). */
76  static std::size_t size ();
77 
78  /** Get a raw pointer to the lookup table. */
79  static const unsigned char* data ();
80 
81  };
