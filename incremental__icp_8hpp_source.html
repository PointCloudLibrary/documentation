47 template <typename PointT, typename Scalar> bool
48 pcl::registration::IncrementalICP<PointT, Scalar>::registerCloud (const PointCloudConstPtr& cloud, const Matrix4& delta_estimate)
49 {
50  assert (icp_);
51 
52  if (!last_cloud_)
53  {
54  last_cloud_ = cloud;
55  abs_transform_ = delta_transform_ = delta_estimate;
56  return (true);
57  }
58 
59  icp_->setInputSource (cloud);
60  icp_->setInputTarget (last_cloud_);
61 
62  {
63  pcl::PointCloud<PointT> p;
64  icp_->align (p, delta_estimate);
65  }
66 
67  bool converged = icp_->hasConverged ();
68 
69  if ( converged ){
70  delta_transform_ = icp_->getFinalTransformation ();
71  abs_transform_ = abs_transform_ * delta_transform_;
72  last_cloud_ = cloud;
73  }
74 
75  return (converged);
76 }
