42 int
43 ParticleFilterTracker<PointInT, StateT>::sampleWithReplacement(
44  const std::vector<int>& a, const std::vector<double>& q)
45 {
46  static std::mt19937 rng{std::random_device{}()};
47  std::uniform_real_distribution<> rd(0.0, 1.0);
48 
49  double rU = rd(rng) * static_cast<double>(particles_->points.size());
50  int k = static_cast<int>(rU);
51  rU -= k; /* rU - [rU] */
52  if (rU < q[k])
53  return k;
54  return a[k];

57 template <typename PointInT, typename StateT>
58 void
59 ParticleFilterTracker<PointInT, StateT>::genAliasTable(
60  std::vector<int>& a,
61  std::vector<double>& q,
62  const PointCloudStateConstPtr& particles)
63 {
64  /* generate an alias table, a and q */
65  std::vector<int> HL(particles->points.size());
66  std::vector<int>::iterator H = HL.begin();
67  std::vector<int>::iterator L = HL.end() - 1;
68  std::size_t num = particles->points.size();
69  for (std::size_t i = 0; i < num; i++)
70  q[i] = particles->points[i].weight * static_cast<float>(num);
71  for (std::size_t i = 0; i < num; i++)
72  a[i] = static_cast<int>(i);
73  // setup H and L
74  for (std::size_t i = 0; i < num; i++)
75  if (q[i] >= 1.0)
76  *H++ = static_cast<int>(i);
77  else
78  *L-- = static_cast<int>(i);
79 
80  while (H != HL.begin() && L != HL.end() - 1) {
81  int j = *(L + 1);
82  int k = *(H - 1);
83  a[j] = k;
84  q[k] += q[j] - 1;
85  ++L;
86  if (q[k] < 1.0) {
87  *L-- = k;
88  --H;
89  }
90  }
91 }

116 template <typename PointInT, typename StateT>
117 void
118 ParticleFilterTracker<PointInT, StateT>::normalizeWeight()
119 {
120  // apply exponential function
121  double w_min = std::numeric_limits<double>::max();
122  double w_max = -std::numeric_limits<double>::max();
123  for (std::size_t i = 0; i < particles_->points.size(); i++) {
124  double weight = particles_->points[i].weight;
125  if (w_min > weight)
126  w_min = weight;
127  if (weight != 0.0 && w_max < weight)
128  w_max = weight;
129  }
130 
131  fit_ratio_ = w_min;
132  if (w_max != w_min) {
133  for (std::size_t i = 0; i < particles_->points.size(); i++) {
134  if (particles_->points[i].weight != 0.0) {
135  particles_->points[i].weight = static_cast<float>(
136  normalizeParticleWeight(particles_->points[i].weight, w_min, w_max));
137  }
138  }
139  }
140  else {
141  for (std::size_t i = 0; i < particles_->points.size(); i++)
142  particles_->points[i].weight =
143  1.0f / static_cast<float>(particles_->points.size());
144  }
145 
146  double sum = 0.0;
147  for (std::size_t i = 0; i < particles_->points.size(); i++) {
148  sum += particles_->points[i].weight;
149  }
150 
151  if (sum != 0.0) {
152  for (std::size_t i = 0; i < particles_->points.size(); i++)
153  particles_->points[i].weight /= static_cast<float>(sum);
154  }
155  else {
156  for (std::size_t i = 0; i < particles_->points.size(); i++)
157  particles_->points[i].weight =
158  1.0f / static_cast<float>(particles_->points.size());
159  }
160 }

338 template <typename PointInT, typename StateT>
339 void
340 ParticleFilterTracker<PointInT, StateT>::resampleWithReplacement()
341 {
342  std::vector<int> a(particles_->points.size());
343  std::vector<double> q(particles_->points.size());
344  genAliasTable(a, q, particles_);
345 
346  const std::vector<double> zero_mean(StateT::stateDimension(), 0.0);
347  // memoize the original list of particles
348  PointCloudStatePtr origparticles = particles_;
349  particles_->points.clear();
350  // the first particle, it is a just copy of the maximum result
351  StateT p = representative_state_;
352  particles_->points.push_back(p);
353 
354  // with motion
355  int motion_num =
356  static_cast<int>(particles_->points.size()) * static_cast<int>(motion_ratio_);
357  for (int i = 1; i < motion_num; i++) {
358  int target_particle_index = sampleWithReplacement(a, q);
359  StateT p = origparticles->points[target_particle_index];
360  // add noise using gaussian
361  p.sample(zero_mean, step_noise_covariance_);
362  p = p + motion_;
363  particles_->points.push_back(p);
364  }
365 
366  // no motion
367  for (int i = motion_num; i < particle_num_; i++) {
368  int target_particle_index = sampleWithReplacement(a, q);
369  StateT p = origparticles->points[target_particle_index];
370  // add noise using gaussian
371  p.sample(zero_mean, step_noise_covariance_);
372  particles_->points.push_back(p);
373  }
374 }

379 {
