53  __device__ __forceinline__ OctreeIteratorDevice(Storage& storage_arg) : storage(storage_arg)
54  {
55  level = 0; // root level
56  storage[level][threadIdx.x] = (0 << 8) + 1;
57  }
58 
59  __device__ __forceinline__ void gotoNextLevel(int first, int len)
60  {
61  ++level;
62  storage[level][threadIdx.x] = (first << 8) + len;
63  }
64 
65  __device__ __forceinline__ int operator*() const
66  {
67  return storage[level][threadIdx.x] >> 8;
68  }
69 
70  __device__ __forceinline__ void operator++()
71  {
72  while(level >= 0)
73  {
74  int data = storage[level][threadIdx.x];
75 
76  if ((data & 0xFF) > 1) // there are another siblings, can goto there
77  {
78  data += (1 << 8) - 1; // +1 to first and -1 from len
79  storage[level][threadIdx.x] = data;
80  break;
81  }
82  else
83  --level; //goto parent;
84  }
85  }
86  };
